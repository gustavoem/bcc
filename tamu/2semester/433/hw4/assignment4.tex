\begin{itemize}
    \item{$V = \{S\}$}
    \item{$\Sigma = \{a, b\}$}

        \subitem{1 $S \rightarrow aSb$}
        \subitem{1 $S \rightarrow Sbb$}\\
    Since these two rules keep $m \geq n$ and $m - n$ even, we have that\\ $R = \{S \rightarrow \epsilon$, $S \rightarrow Sbb$, $S \rightarrow aSb\}$.
