from sklearn.model_selection import GridSearchCV

# Given a set of size N, returns a random partition of this set
# containing two subsets of size N - k and k
def random_bipartite_data (data_set, k):
    np.random.shuffle (data_set)
    return (data_set[k:], data_set[:k])


# Plots E_in and E_val
def plot_error (n, errors_list, labels_list):
    ax = plt.gca ()
    x = range (1, n + 1)
    for i, error_t in enumerate (errors_list):
        plt.plot (x, error_t, label = labels_list[i])    
    handlers, labels = ax.get_legend_handles_labels ()
    ax.legend (handlers, labels, loc = 'upper right', numpoints = 1)
    plt.ylabel ('Relative error')
    plt.xlabel ('Number of images sampled x500')
    plt.show ()
    plt.clf ()

Classificação de dígitos¶

Os dataset para esta tarefa uma adaptao do disponvel na competio do kaggle de reconhecimento de dgitos (https://www.kaggle.com/c/digit-recognizer) e est disponvel em http://vision.ime.usp.br/~caiomr/mac0460_5832/train_svm.csv.gz. O dataset est sob a licena Creative Commons Attribution-Share Alike 3.0 license (https://creativecommons.org/licenses/by-sa/3.0/). O dataset foi zipado, e apenas os dgitos 5 e 6 foram mantidos. Cada linha (amostra) do arquivo contm 257 colunas: a primeira informa o label da amostra (0 para o dgito 5, 1 para o dgito 6) e as outras 256 so os valores dos pixels da imagem (16 x 16) que representa o dgito.

Note que esse dataset difere do usado no EP3: as imagens sofreram pequenas rotaes e translaes aleatrias, alm de terem sido escalonadas para o tamanho 16x16. Veja tambm que pode ser necessrio realizar algum tipo de normalizao para realizar um treinamento efetivo com SVM. Para auxiliar na normalizao dos dados, consultem o seguinte link: http://scikit-learn.org/stable/modules/preprocessing.html.

Q1. Use SVM para classificar os dgitos 5 e 6. Utilize as funes do scikit learn (http://scikit-learn.org/, http://scikit-learn.org/stable/modules/svm.html) para realizar o treinamento.

Teste os kernels linear e RBF da seguinte maneira:

  1. Escolha aleatoriamente 932 amostras para formarem o conjunto de teste.
  2. Com as 7000 amostras restantes, utilize validao cruzada (com nmero de folds K = 5) para escolher os parmetros do seu classificador, isto : C (peso da soft margin) para o kernel linear; C e gamma para o kernel RBF.
  3. Plote a curva experimental de aprendizado para o melhor SVM com kernel linear e o melhor SVM com kernel RBF escolhidos por validao cruzada. Use as 932 amostras do conjunto de teste para estimar $E_{out}$. Comente sobre o resultado.

linear_svm = SVC (C=10, kernel='linear')

e_in_arr = []
e_out_arr = []
e_out = 0

for i in range (0, validation_data.shape[0], 500):
    linear_svm.fit (validation_data[i:i + 500, 1:], 
                    validation_data[i:i + 500, 0])
    e_in = 1 - linear_svm.score (validation_data[:i + 500, 1:], 
                             validation_data[:i + 500, 0])
    e_out = 1 - linear_svm.score (test_data[:, 1:],
                              test_data[:, 0])
    e_in_arr.append (e_in)
    e_out_arr.append (e_out)
        
labels_list = ['$E_{in}$', '$E_{out}$']        
erros_list = [e_in_arr, e_out_arr]
print ("Erro para o melhor classificador com kernel linear:")
plot_error (len (e_in_arr), erros_list, labels_list)
print ("E_out = " + str (e_out))
